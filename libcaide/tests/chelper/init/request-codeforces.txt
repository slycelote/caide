    String.prototype.hashCode = function() {

                        var text = $(i).val();
                        if (text) {

                                    found = true;
                                }
                            }
                            if ($(this).hasClass('visible')) {
                                visibleRowCount++;
                            }
                        });

                            parent.find(".rowCount").text("Matches: " + (visibleRowCount - (found ? 1 : 0)));

                        if (visibleRowCount == (found ? 1 : 0)) {
